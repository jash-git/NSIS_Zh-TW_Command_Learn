C.8 判斷 Macromedia Flash Player 是否已安裝

 ; IsFlashInstalled
 ;
 ; Yazno, http://yazno.tripod.com/powerpimpit/
 ; 返回值到堆棧頂
 ; 0 (Flash 未安裝)
 ; 或
 ; 1 (Flash 已安裝)
 ;
 ; 用法:
 ;   Call IsFlashInstalled
 ;   Pop $R0
 ;   ; $R0 在這裡為 "1" 或 "0"

 Function IsFlashInstalled
  Push $R0
  ClearErrors
  ReadRegStr $R0 HKCR "CLSID\{D27CDB6E-AE6D-11cf-96B8-444553540000}" ""
  IfErrors lbl_na
    StrCpy $R0 1
  Goto lbl_end
  lbl_na:
    StrCpy $R0 0
  lbl_end:
  Exch $R0
 FunctionEnd
